l = 4
e = 0.25
c = (8*e)/pi**2
x = np.linspace(0,l)
y1 = c*np.sin((pi*x)/l)
y3 = c*(-1/9)*np.sin((3*pi*x)/l)
y5 = c*(1/25)*np.sin((5*pi*x)/l)
y7 = c*(-1/49)*np.sin((7*pi*x)/l)
y9 = c*(1/81)*np.sin((9*pi*x)/l)
y11 = c*(-1/121)*np.sin((11*pi*x)/l)
yt = y1+y3+y5+y7+y9+y11
